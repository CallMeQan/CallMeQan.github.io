## Approach 1: Recursive

- Time Complexity: **$O(2^n)$**
- Memory Complexity: **$O(n)$**
#### Code Sample
```cpp
int fib(int num){
	if(num <= 0) return 0;
	else if(num <= 2) return 1;
	else{
		return fib(num - 1) + fib(num - 2);
	} 
}

int main(){
	int n;
	cin>>n;
	cout<<fib(n)<<endl;
}

```
## Approach 2: Memorization ([[Dynamic Programming]])

- Time Complexity: **$O(n)$**
- Memory Complexity: **$O(n)$**
#### Code Sample

```cpp
int main(){
	int n;
	cin>>n;

	int dp[n+1];
	dp[0] = 0;
	dp[1] = 1;
	dp[2] = 1;
	for(int i = 3; i <= n; i++){
		dp[i] = dp[i-1] + dp[i-2];
	}
	cout<<dp[n]<<endl;
	return 0;
}
```

## Approach 3: [[Matrix Exponentiation]] (Fastest)

- Time Complexity: $O(log(n))$
- Memory Complexity: $O(1)$
### Code Sample
##### Cpp
```cpp
// Generated by ChatGPT 4
#include <bits/stdc++.h>
using namespace std;

const int MOD = 1e9 + 7; // Optional: Modular arithmetic for large Fibonacci numbers.

struct Matrix {
    long long mat[2][2];

    // Constructor to initialize the matrix
    Matrix(long long a = 1, long long b = 1, long long c = 1, long long d = 0) {
        mat[0][0] = a; mat[0][1] = b;
        mat[1][0] = c; mat[1][1] = d;
    }

    // Overload the multiplication operator for matrix multiplication
    Matrix operator*(const Matrix &other) const {
        Matrix result;
        result.mat[0][0] = (mat[0][0] * other.mat[0][0] + mat[0][1] * other.mat[1][0]) % MOD;
        result.mat[0][1] = (mat[0][0] * other.mat[0][1] + mat[0][1] * other.mat[1][1]) % MOD;
        result.mat[1][0] = (mat[1][0] * other.mat[0][0] + mat[1][1] * other.mat[1][0]) % MOD;
        result.mat[1][1] = (mat[1][0] * other.mat[0][1] + mat[1][1] * other.mat[1][1]) % MOD;
        return result;
    }
};

// Function to perform matrix exponentiation
Matrix matrixExpo(Matrix base, long long exp) {
    Matrix result(1, 0, 0, 1); // Identity matrix
    while (exp > 0) {
        if (exp % 2 == 1) {    // If the exponent is odd
            result = result * base;
        }
        base = base * base;    // Square the base
        exp /= 2;
    }
    return result;
}

// Function to calculate the nth Fibonacci number
long long fibonacci(long long n) {
    if (n == 0) return 0;  // Base case
    if (n == 1) return 1;  // Base case

    Matrix fibMatrix(1, 1, 1, 0);
    Matrix result = matrixExpo(fibMatrix, n - 1);

    return result.mat[0][0]; // The top-left element gives F(n)
}

int main() {
    long long n;
    cout << "Enter the value of n: ";
    cin >> n;

    cout << "F(" << n << ") = " << fibonacci(n) << endl;

    return 0;
}
```

##### Python:
```python
MOD = 10**9 + 7  # Optional: Modular arithmetic for large Fibonacci numbers.

class Matrix:
    def __init__(self, a=1, b=1, c=1, d=0):
        self.mat = [[a, b], [c, d]]

    def __mul__(self, other):
        """Overload the multiplication operator for matrix multiplication."""
        a = self.mat[0][0] * other.mat[0][0] + self.mat[0][1] * other.mat[1][0]
        b = self.mat[0][0] * other.mat[0][1] + self.mat[0][1] * other.mat[1][1]
        c = self.mat[1][0] * other.mat[0][0] + self.mat[1][1] * other.mat[1][0]
        d = self.mat[1][0] * other.mat[0][1] + self.mat[1][1] * other.mat[1][1]
        return Matrix(a % MOD, b % MOD, c % MOD, d % MOD)

def matrix_exponentiation(base, exp):
    """Perform matrix exponentiation using exponentiation by squaring."""
    result = Matrix(1, 0, 0, 1)  # Identity matrix
    while exp > 0:
        if exp % 2 == 1:  # If the exponent is odd
            result = result * base
        base = base * base  # Square the base
        exp //= 2
    return result

def fibonacci(n):
    """Calculate the nth Fibonacci number."""
    if n == 0:
        return 0
    if n == 1:
        return 1

    fib_matrix = Matrix(1, 1, 1, 0)
    result_matrix = matrix_exponentiation(fib_matrix, n - 1)
    return result_matrix.mat[0][0]  # The top-left element gives F(n)

# Example usage
if __name__ == "__main__":
    n = int(input("Enter the value of n: "))
    print(f"F({n}) = {fibonacci(n)}")
```
